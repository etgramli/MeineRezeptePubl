\begin{recipe}[
    preparationtime = {\unit[10]{Min}},
    portion = {\portion[Glas]{1}},
    source = Lili
]{Nutelili}\index{Nutelili}
    \graph{
        big=Nutelili_0
    }
    \ingredients{
        {\unit[60]{g}}       & Butter\\
        {\unit[50]{g}}       & Schokolade (85\%)\\
        {\unit[120]{g}}      & Haselnüsse, geröstet\\
        {\unit[50]{g}}       & Erythrit, gemahlen\\
        {\unit[1]{Prise}}    & Vanille\\
        {\unit[5]{g}}        & Kakao\\
        {\unit[60]{g}}       & Sahne
    }
    \preparation{
        \step Die Butter und Schokolade bei niedriger Hitze schmelzen.
        \step Die Nüsse, Erythrit, Vanille und Kakao mahlen, die restlichen Zutaten dazu geben und vermischen.
    }
    \hint{%
        Mit der doppelten Menge Sahne wird es streichzart (für Weißbrot).
    }
\end{recipe}
