\begin{recipe}[
    preparationtime = {\unit[15]{Min}},
    bakingtime = {\unit[1]{h}},
    bakingtemperature = {\unit[160]{\celsius}},
    source = Lili
]{Ketogener Marmorkuchen}\index{Ketogener Marmorkuchen}
    \graph{
        big=KetogenerMarmorkuchen_0
    }
    \ingredients{
        {\unit[80]{g}}       & Butter, geschmolzen\\
        {\unit[200]{g}}      & Joghurt\\
        3                    & Eier\\
        {\unit[50]{g}}       & Mandelmehl\\
        {\unit[150]{g}}      & Mandeln, gemahlen\\
        {\unit[2]{EL}}       & Kakao\\
        {\unit[100]{g}}      & Xylith\\
        {\unit[1]{EL}}       & Inulin\\
        {\unit[8]{g}}        & Backpulver\\
        {\unit[1]{Prise}}    & Vanille\\
        {\unit[1]{Prise}}    & Salz
    }
    \preparation{
        \step Alle feuchten Zutaten mischen, alle trockenen Zutaten (außer Kakao) mischen und beides verrühren.
        \step Den Teig in zwei Hälften teilen und unter eine Hälfte den Kakao unterrühren. Beide Hälften unterheben.
        \step Teig in eine 18-er-Form geben und im vorgeheizten Backofen eine~Stunde bei {\unit[160]{\celsius}} backen.
    }
\end{recipe}
