\begin{recipe}[
    portion = {\portion{4}},
    source = Lili
]{Waffeln}\index{Waffeln}
    \graph{
        big=Waffeln_0
    }
    \ingredients{
        {\unit[125]{g}}      & Butter, weich\\
        {\unit[50]{g}}       & Zucker\\
        4                    & Eier\\
        {\unit[2]{EL}}       & Vanille-Zucker\\
        {\unit[1]{Prise}}    & Salz\\
        {\unit[250]{g}}      & Mehl\\
        {\unit[1]{TL}}       & Backpulver\\
        {\unit[250]{ml}}     & Sahne\\
                             & Öl
    }
    \preparation{
        \step Die Butter, Zucker, Vanille-Zucker, Salz und Eier schaumig rühren.
        \step Mehl mit dem Backpulver nach und nach abwechselnd mit der Sahne unter den Teig rühren.
        \step So viel Sahne zugeben, dass ein dünnflüssiger Teig entsteht.
    }
    \hint{%
        Immer wieder, weils gut schmeckt
    }
\end{recipe}
