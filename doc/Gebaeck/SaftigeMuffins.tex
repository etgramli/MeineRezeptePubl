    }
    \ingredients{
        {\unit[120]{g}}     & Butter\\
        {\unit[110]{g}}     & Zucker\\
        {\unit[1]{P.}}      & Vanille-Zucker\\
        {\unit[80]{g}}      & Naturjoghurt\\
        {\unit[120]{g}}     & Mehl\\
        {\unit[1]{TL}}      & Backpulver\\
                            & Zitronenaroma\\
        2                   & Eier\\
        {\unit[1]{Dose}}    & Mandarinen
    }
    \preparation{
        \step Butter und Zucker mischen, mit Eiern und Joghurt glatt rühren; Mehl und Backpulver unterrühren.

        \step Im vorgeheizten Backofen bei {\unit[180]{\celsius}} 20-25 Minuten backen.
    }
