\begin{recipe}[
    preparationtime = {\unit[10]{Min}},
    portion = {\portion[Glas]{1}},
    source = Etienne
]{Pesto Rosso}\index{Pesto Rosso}\index{Pesto!Pesto Rosso}
    \graph{
        small=PestoRosso_1,
        big=PestoRosso_0
    }
    \ingredients{
        {\unit[125]{g}}                   & Tomaten, getrocknet, ohne Öl\\
        {\unit[50]{g}}                    & Parmesan\\
        {\unit[50]{g}}                    & Pinienkerne\\
        {\unit[$\nicefrac{1}{2}$]{TL}}    & Salz\\
        {\unit[1]{Prise}}                 & Pfeffer\\
        {\unit[2]{EL}}                    & Rotwein\\
        {\unit[125]{g}}                   & Olivenöl
    }
    \preparation{
        \step Die Pinienkerne anrösten, währenddessen den Parmesan 10 Sekunden auf Stufe 10 zerkleinern.
        \step Die Tomaten, Pinienkerne, Salz und Pfeffer dazu geben und 20 Sekunden auf Stufe 7 zerkleinern.
        \step Den Rotwein dazu geben und unter Rühren das Öl untermischen.
        \step Auf Brot oder Nudeln servieren oder in kleine Gläschen mit etwas Öl darauf abfüllen.
    }
\end{recipe}
