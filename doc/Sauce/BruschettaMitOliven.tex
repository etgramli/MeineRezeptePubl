\begin{recipe}[
    preparationtime = {\unit[10]{Min}},
    portion = {\portion[Glas]{1}},
    source = Etienne
]{Bruschetta mit Oliven}\index{Bruschetta mit Oliven}
    \graph{
        big=BruschettaMitOliven_0
    }
    \ingredients{
        {\unit[180]{g}}    & Oliven, schwarz, entsteint\\
        1                  & Knoblauchzehe\\
        {\unit[35]{g}}     & Parmesan\\
        2                  & Tomaten, getrocknet, in Öl\\
        {\unit[1]{EL}}     & Olivenöl\\
                           & Salz\\
                           & Pfeffer\\
                           & Oregano
    }
    \preparation{
        \step Parmesan 8~Sekunden auf Stufe~10 zerkleinern, Oliven und Tomaten dazu geben und bei Stufe~7 hacken.
        \step Die Knoblauchzehe dazu pressen, Olivenöl und Gewürze dazu geben und vermischen.
        \step Auf Brot oder Nudeln servieren oder in kleine Gläschen mit etwas Öl darauf abfüllen.
    }
\end{recipe}
