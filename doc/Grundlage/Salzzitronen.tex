\begin{recipe}[
    preparationtime = {\unit[5]{Min}},
    source = Afafe
]{Salzzitronen}\index{Salzzitronen}
    \graph{
        big=Salzzitronen_0
    }
    \ingredients{
        2                  & Zitronen (Bio)\\
        {\unit[250]{g}}    & Salz
    }
    \preparation{
        \step Von den Zitronen den Stängel entfernen, gut waschen und abtrocknen.
        \step Die Zitronen längs vierteln und mit Salz abwechselnd in ein Glas schichten.
        \step Ein paar Wochen ziehen lassen damit die Schale weich wird.
    }
    \hint{%
        Man sollte das Glas nicht ganz füllen, weil das Salz den Saft aus den Zitronen zieht und sich ausdehnt.\newline
        Ein Viertel passt gut in eine Tajine mit Hühnchen.\newline
        Das Salz kann man z.B. für Tomaten- und Salatsaucen benutzen.
    }
\end{recipe}
