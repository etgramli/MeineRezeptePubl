\begin{recipe}[
    portion = {\portion[Portionen]{4}},
    source = Etienne
]{Aprikosen-Sahne-Dessert}\index{Aprikosen-Sahne-Dessert}
    \graph{
        big=AprikosenSahneDessert_0
    }
    \ingredients{
        14                  & Aprikosen, getrocknet\\
        {\unit[40]{ml}}     & Straight Bourbon\\
        {\unit[4]{TL}}      & Honig\\
        {\unit[2]{B.}}      & Sahne\\
        {\unit[1]{B.}}      & Naturjoghurt\\
        {\unit[4]{TL}}      & Zucker\\
        {\unit[2]{TL}}      & Zucker
    }
    \preparation{
        \step Den Honig im Bourbon auflösen und die Aprikosen darin über Nacht einlegen. Dann pürieren.
        \step Die Sahne schlagen und mit Joghurt vermischen. Vier Gläser damit zur Hälfte füllen.
        \step Aprikosen darauf geben, den Rest der Sahne darauf verteilen und mit etwas Zucker bedecken.
        \step Den Zucker leicht flambieren, bis er goldbraun ist und sofort servieren.
    }
\end{recipe}
