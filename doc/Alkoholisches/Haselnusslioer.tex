\begin{recipe}[
    preparationtime = {\unit[15]{Min}},
    portion = {\portion[ml]{500}},
    source = Etienne
]{Haselnusslikör}\index{Haselnusslikoer@Haselnusslikör}
    \graph{
        big=Haselnusslikoer_0,
        small=Haselnusslikoer_1
    }
    \ingredients{
        {\unit[200]{g}}     & Haselnusskerne\\
        {\unit[220]{ml}}    & Primaspirit (\unit[70]{\%})\\
        {\unit[280]{ml}}    & Wasser, destilliert\\
        {\unit[40]{g}}      & Zucker
    }
    \preparation{
        \step Die Haselnusskerne in einer Pfanne leicht rösten und mit dem Zucker in eine Flasche füllen.
        \step Mit Primaspirit und Wasser auffüllen und etwa 2~Monate ziehen lassen.
        \step Den Likör durch ein Sieb in eine \unit[500]{ml}-Flasche umfüllen und dunkel lagern.
    }
\end{recipe}
