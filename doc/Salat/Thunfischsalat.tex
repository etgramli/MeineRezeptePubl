\begin{recipe}[
    preparationtime = {\unit[5]{Min}},
    portion = {\portion{1}},
    source = Etienne
]{Thunfischsalat}\index{Thunfischsalat}
    \graph{
        big=Thunfischsalat_0
    }
    \ingredients{
        {\unit[1]{Dose}}     & Thunfischfilet, in Öl\\
        $\nicefrac{1}{2}$    & Zwiebel, klein\\
        6                    & Oliven, schwarz\\
        {\unit[1]{EL}}       & Petersilie\\
        etwas                & Zitronensaft\\
        {\unit[1]{Prise}}    & Salz\\
                             & Pfeffer
    }
    \preparation{
        \step Die halbe Zwiebel in sehr feine Würfel schneiden, die Oliven halbieren und den Fisch aus dem Öl nehmen.
        \step Thunfisch, Zwiebel, Oliven und Petersilie vermischen und mit Zitronensaft, Salz und Pfeffer abschmecken.
    }
    \hint{%
        Statt der Petersilie kann man eine Prise getrockneten Oregano hinzufügen.
    }
\end{recipe}
