\begin{recipe}[
    preparationtime = {\unit[5]{Min}},
    portion = {\portion{3}},
    source = Etienne
]{Bauernsalat}\index{Bauernsalat}
    \graph{
        big=Bauernsalat_0
    }
    \ingredients{
        1                  & Gurke\\
        1                  & Fleischtomate\\
        1                  & Zwiebel, rot, klein\\
        10                 & Oliven, schwarz\\
        {\unit[200]{g}}    & Schafskäse\\
        {\unit[1]{EL}}     & Essig\\
        {\unit[2]{EL}}     & Öl\\
        {\unit[1]{EL}}     & Liebstöckel\\
        {\unit[1]{EL}}     & Petersilie\\
        {\unit[1]{TL}}     & Salz\\
                           & Pfeffer
    }
    \preparation{
        \step Gurke längs zwei mal einschneiden und in viertel Räder schneiden; Tomaten in kleine Schnitze schneiden.
        \step Oliven halbieren und Zwiebel und Schafskäse in Würfel schneiden und unterheben.
        \step Liebstöckel und Petersilie hacken. Mit Salz und Pfeffer würzen und mit Essig und Öl abschmecken.
    }
\end{recipe}
