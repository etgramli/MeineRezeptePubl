\begin{recipe}[
    preparationtime = {\unit[15]{Min}},
    portion = {\portion[Portionen]{3}},
    source = Etienne
]{Linsensalat}\index{Linsensalat}
    \graph{
        big=Linsensalat_0
    }
    \ingredients{
        {\unit[250]{g}}        & Linsen\\
        $\nicefrac{1}{2}$      & Zwiebel\\
        $\nicefrac{1}{4}$      & Chilischote\\
        etwas                  & Zitronenschale\\
        {\unit[1]{EL}}         & Petersilie\\
        {\unit[4]{Spitzen}}    & Liebstöckel\\
        {\unit[3]{EL}}         & Aceto Balsamico\\
        {\unit[4]{EL}}         & Öl\\
                               & Salz\\
                               & Pfeffer
    }
    \preparation{
        \step Die Linsen über Nacht in Wasser ziehen lassen; am nächsten Tag mit Zwiebelwürfeln weich kochen.
        \step Chili klein schneiden, zu den Linsen geben, Zitronenschale dazu reiben und die Kräuter darüber streuen.
        \step Mit Salz und Pfeffer abschmecken und mit Essig und Öl fertig stellen.
    }
\end{recipe}
