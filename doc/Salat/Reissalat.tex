\begin{recipe}[
    preparationtime = {\unit[15]{Min}},
    portion = {\portion{6}},
    source = Etienne
]{Reissalat}\index{Reissalat}
    \graph{
        big=Reissalat_0
    }
    \ingredients{
        {\unit[1]{Tasse}}    & Reis\\
        8                    & Tomaten, getrocknet\\
        3                    & Pepperoni\\
        12                   & Oliven\\
        {\unit[100]{g}}      & Schafkäse\\
        {\unit[1]{Glas}}     & Pesto\\
        {\unit[1]{Bund}}     & Rucola
    }
    \preparation{
        \step Den Reis kochen, währenddessen die Tomaten, Pepperoni, Oliven und Schafkäse schneiden.
        \step Reis mit den restlichen Zutaten vermischen, servieren und mit dem Rucola garnieren.
    }
\end{recipe}
