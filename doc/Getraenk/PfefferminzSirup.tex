\begin{recipe}[
    preparationtime = {\unit[15]{Min}},
    bakingtime = {\unit[10]{Min}},
    portion = {\portion[l]{$1\nicefrac{1}{2}$}},
    source = Lili
]{Pfefferminz-Sirup}\index{Pfefferminz-Sirup}
    \graph{
        big=PfefferminzSirup_0
    }
    \ingredients{
        {\unit[10]{Stängel}}    & Pfefferminze\\
        {\unit[30]{g}}          & Zitronensäure\\
        {\unit[1]{l}}           & Wasser, kalt\\
        {\unit[700]{g}}         & Zucker
    }
    \preparation{
        \step Zitronensäure in Wasser auflösen, Blätter dazugeben und 24 Stunden stehen lassen.
        \step Aufkochen, absieben und Zucker darin auflösen; ca. 10 Minuten einkochen.
        \step Noch heiß in Flaschen füllen und auf den Kopf stellen.
    }
    \hint{%
        Funktioniert auch gut im Thermomix mit Varoma.
    }
\end{recipe}
