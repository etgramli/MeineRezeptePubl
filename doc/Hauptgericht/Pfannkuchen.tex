        {\unit[50]{g}}       & Vollkornmehl\\
        {\unit[300]{ml}}     & Milch\\
        {\unit[1]{Prise}}    & Salz\\
        4                    & Eier\\
        {\unit[2]{EL}}       & Butter, flüssig
    }
