\begin{recipe}[
    preparationtime = {\unit[20]{Min}},
    portion = {\portion{2}},
    source = Etienne
]{Spaghetti mit Zucchini und Rübe}\index{Spaghetti mit Zucchini und Ruebe@Spaghetti mit Zucchini und Rübe}
    \graph{
        big=SpaghettiMitZucchiniUndRuebe_0
    }
    \ingredients{
        {\unit[300]{g}}                   & Spaghetti\\
        1                                 & Zucchini, groß\\
        1                                 & Rübe\\
        {\unit[2]{EL}}                    & Olivenöl\\
        {\unit[1]{EL}}                    & Pesto\\
        {\unit[$\nicefrac{1}{4}$]{TL}}    & Salz\\
        {\unit[1]{Prise}}                 & Pfeffer\\
                                          & Parmesan
    }
    \preparation{
        \step Die Spaghetti kochen; währenddessen die Zucchini waschen und abtrocknen, die Rübe schälen.
        \step Die Zucchini und und Rübe in eine Pfanne hobeln und in Olivenöl andünsten.
        \step Mit Salz und Pfeffer würzen, die gekochten Spaghetti darüber gießen, Pesto dazu geben und vermischen.
        \step Spaghetti in einen Teller geben und mit geriebenem Parmesan garnieren.
    }
\end{recipe}
