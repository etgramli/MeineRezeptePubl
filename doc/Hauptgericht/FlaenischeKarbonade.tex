\begin{recipe}[
    preparationtime = {\unit[30]{Min}},
    bakingtime = {\unit[90]{Min}},
    portion = {\portion{4}},
    source = Chefkoch
]{Flänische Karbonade}\index{Flaenische Karbonade@Flänische Karbonade}
    \graph{
        big=FlaenischeKarbonade_0
    }
    \ingredients{
        {\unit[1]{kg}}         & Rindergulasch\\
        {\unit[2]{EL}}         & Butter\\
        2                      & Zwiebeln\\
        {\unit[1]{TL}}         & Tomatenmark\\
        3                      & Knoblauchzehen\\
        {\unit[1]{Stück}}      & Sellerie\\
        {\unit[1]{Stange}}     & Lauch\\
        1                      & Lorbeerblatt\\
        {\unit[1]{Zweig}}      & Thymian\\
        1                      & Gewürznelke\\
        {\unit[400]{ml}}       & Bier\\
                               & Salz\\
                               & Pfeffer
    }
    \preparation{
        \step Fleisch in der Butter scharf anbraten, gewürfelte Zwiebeln dazu geben und andünsten.
        \step Tomatenmark unterrühren und Knoblauch kurz anschwitzen und mit Bier ablöschen.
        \step Sellerie und Lauch schneiden und mit Lorbeerblatt, Thymian und Nelke dazu geben.
        \step Bei mittlerer Hitze $1\nicefrac{1}{2}$ bis 2~Stunden schmoren, mit Salz und Pfeffer abschmecken und evt. eindicken.
    }
\end{recipe}
