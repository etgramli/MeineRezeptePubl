\begin{recipe}[
    preparationtime = {\unit[20]{Min}},
    bakingtime = {\unit[4]{Min}},
    bakingtemperature = {\unit[150]{\celsius}},
    portion = {\portion{2}},
    source = Etienne
]{Lachsrollen}\index{Lachsrollen}
    \graph{
        small=Lachsrollen_1,
        big=Lachsrollen_0
    }
    \ingredients{
        {\unit[200]{g}}        & Lachs, geräuchert und geschnitten\\
        1                      & Zwiebel\\
        {\unit[360]{g}}        & Spinat\\
        {\unit[300]{g}}        & Frischkäse\\
        {\unit[1]{Schuss}}     & Zitronensaft\\
        {\unit[4]{Blätter}}    & Filo-Teig\\
                               & Salz\\
                               & Pfeffer
    }
    \preparation{
        \step Zwiebel würfeln, glasig dünsten; Spinat dazu geben, auch andünsten und mit Salz und Pfeffer würzen.
        \step Frischkäse und Zitronensaft vermischen und dünn auf dem Filo-Teig bis zum Rand verstreichen.
        \step Den Teig 4~Minuten im vorgeheizten Ofen backen, Lachs und Spinat darauf verteilen und aufrollen.
    }
\end{recipe}
