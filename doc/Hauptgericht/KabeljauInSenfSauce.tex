\begin{recipe}[
    preparationtime = {\unit[20]{Min}},
    portion = {\portion{2}},
    source = Etienne
]{Kabeljau in Senfsauce}\index{Kabeljau in Senfsauce}\index{Senfsauce}
    \graph{
        big=KabeljauInSenfSauce_0
    }
    \ingredients{
        {\unit[225]{g}}       & Kabeljaufilet\\
        {\unit[100]{g}}       & Reis\\
        {\unit[300]{g}}       & Sahne\\
        {\unit[1]{EL}}        & Senf, mittelscharf\\
        {\unit[2]{EL}}        & Dill\\
        {\unit[1]{Prise}}     & Muskat\\
        {\unit[1]{Schuss}}    & Zitronensaft\\
                              & Salz\\
                              & Pfeffer, schwarz
    }
    \preparation{
        \step Wasser zum Kochen bringen und den Reis 10~Minuten kochen.
        \step Währenddessen den Kabeljau trocken tupfen und mit Salz, Pfeffer und Mehl bestäuben.
        \step Den Kabelau in einer Pfanne anbraten. Sobald er bräunt rausnehmen und zur Stei stellen.
        \step Die Sahne in die Pfanne geben und kurz aufkochen lassen und Senf darin auflösen.
        \step Dill, Muskat und Zitronensaft dazu geben und mit Salz und Pfeffer abschmecken.
        \step Den Fisch kurz in der Sauce ziehen lassen, den Reis abgießen und alles zusammen servieren.
    } 
    \hint{%
        Dazu passen gut Erbsen.
    }
\end{recipe}
