\begin{recipe}[
    preparationtime = {\unit[20]{Min}},
    portion = {\portion{4}},
    source = Lili
]{Spaghetti Carbonara}\index{Spaghetti Carbonara}
    \graph{
        big=SpaghettiCarbonara_0
    }
    \ingredients{
        {\unit[400]{g}}     & Spaghetti\\
        {\unit[200]{g}}     & Speck, geräuchert\\
        {\unit[400]{ml}}    & Sahne\\
        1                   & Knoblauchzehe\\
        {\unit[50]{g}}      & Parmesan\\
                            & Olivenöl\\
                            & Salz\\
                            & Pfeffer, schwarz und weiß
    }
    \preparation{
        \step Spaghetti kochen und währenddessen Speck und Knoblauch in Öl anbraten.
        \step Mit der Sahne ablöschen, kurz aufkochen und Parmesan dazu reiben. Mit Salz und Pfeffer abschmecken.
        \step Spaghetti abgießen, kurz in der Sauce ziehen lassen und servieren.
    }
\end{recipe}
