\begin{recipe}[
    preparationtime = {\unit[15]{Min}},
    portion = {\portion{4}},
    source = Lili
]{Spaghetti Carbonara}\index{Spaghetti Carbonara}
    \graph{
        big=SpaghettiCarbonara_0
    }
    \ingredients{
        {\unit[400]{g}}     & Spaghetti\\
        {\unit[200]{g}}     & Schinken, geräuchert und gekocht\\
        {\unit[200]{ml}}    & Creme Fraîche\\
        {\unit[200]{ml}}    & Sahne\\
        1                   & Knoblauchzehe\\
                            & Olivenöl\\
                            & Salz\\
                            & Pfeffer\\
                            & Kräuter
    }
    \preparation{
        \step Spaghetti kochen und währenddessen Schinken und Knoblauch in Öl anbraten.
        \step Kräuter dazu geben, mit Wasser oder Brühe und Sahne ablöschen und zum köcheln bringen.
        \step Spaghetti dazu geben, abkühlen lassen und servieren.
    }
\end{recipe}
