\begin{recipe}[
    preparationtime = {\unit[20]{Min}},
    portion = {\portion{4}},
    source = Lili
]{Spaghetti Carbonara}\index{Spaghetti Carbonara}
    \graph{
        big=SpaghettiCarbonara_0
    }
    \ingredients{
        {\unit[400]{g}}     & Spaghetti\\
        {\unit[200]{g}}     & Bauchspeck, geräuchert\\
        {\unit[400]{ml}}    & Sahne\\
        1                   & Knoblauchzehe\\
                            & Olivenöl\\
                            & Salz\\
                            & Pfeffer, schwarz und weiß
    }
    \preparation{
        \step Spaghetti kochen und währenddessen Schinken und Knoblauch in Öl anbraten.
        \step Mit der Sahne ablöschen, kurz aufkochen und mit Salz und Pfeffer abschmecken.
        \step Spaghetti abgießen, kurz in der Sauce ziehen lassen und servieren.
    }
\end{recipe}
