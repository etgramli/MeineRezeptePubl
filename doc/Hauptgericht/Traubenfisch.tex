\begin{recipe}[
    bakingtime = {\unit[20]{Min}},
    portion = {\portion{2}},
    source = Etienne
]{Traubenfisch}\index{Traubenfisch}
    \graph{
        big=Traubenfisch_0
    }
    \ingredients{
        2                 & Kabeljaufilets\\
        1                 & Rübe, groß\\
        1                 & Zucchini\\
        1                 & Knoblauchzehe\\
        {\unit[4]{EL}}    & Verjus\\
                          & Salz\\
                          & Pfeffer
    }
    \preparation{
        \step Rübe schälen, Zucchini waschen und beide in eine Pfanne hobeln und in Öl andünsten.
        \step Kabeljau von beiden Seiten salzen, von nur einer pfeffern und in der Pfanne anbraten.
        \step Sobald Kabeljau und Rübe bräunen mit dem Verjus ablöschen und mit Salz und Pfeffer abschmecken.
    }
\end{recipe}
