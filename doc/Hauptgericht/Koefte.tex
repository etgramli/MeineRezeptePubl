\begin{recipe}[
    preparationtime = {\unit[20]{Min}},
    portion = {\portion{2}},
    source = Müge
]{Köfte}\index{Koefte@Köfte}
    \graph{
        big=Koefte_0
    }
    \ingredients{
        {\unit[500]{g}}    & Hackfleisch\\
        1                  & Zwiebel, mittelgroß\\
        1                  & Ei\\
        {\unit[1]{TL}}     & Kreuzkümmel, gemahlen\\
        {\unit[1]{TL}}     & Thymian\\
        {\unit[1]{EL}}     & Petersilie\\
                           & Chiliflocken\\
                           & Salz
    }
    \preparation{
        \step Die Zwiebel in kleine Würfel schneiden und mit dem Hackfleisch, Ei und den Gewürzen vermischen.
        \step Kleine Küchlein formen und in einer Pfanne mit heißem Öl braten.
    }
    \hint{%
        Reis passt gut dazu.
    }
\end{recipe}
