\begin{recipe}[
    preparationtime = {\unit[30]{Min}},
    portion = {\portion{2}},
    source = Etienne
]{Yufka}\index{Yufka}
    \graph{
        big=Yufka_0
    }
    \ingredients{
        {\unit[2]{Scheiben}}    & Rinderrouladen\\
        2                       & Fladen / Wraps\\
        {\unit[1]{TL}}          & Thymian\\
        {\unit[1]{TL}}          & Paprikapulver\\
        {\unit[1]{Prise}}       & Koriander\\
        {\unit[1]{Prise}}       & Pfeffer\\
        {\unit[4]{EL}}          & Joghurt\\
        {\unit[2]{EL}}          & Petersilie\\
        {\unit[1]{Zweig}}       & Liebstöckel\\
        {\unit[1]{Prise}}       & Salz\\
        1                       & Tomate\\
        $\nicefrac{1}{2}$       & Zwiebel, klein\\
        $\nicefrac{1}{4}$       & Chili\\
        $\nicefrac{1}{4}$       & Gurke
    }
    \preparation{
        \step Chili und Tomate in Ringe schneiden und die Gurke in dünne Scheiben hobeln.
        \step Den Joghurt mit Petersilie und Liebstöckel vermischen, die Zwiebel in Ringe schneiden und salzen.
        \step Die Fladen von beiden Seiten in der Pfanne backen bis sich Blasen bilden.
        \step Rinderrouladen in dünne Streifen schneiden und mit Thymian, Paprikapulver, Koriander, Pfeffer würzen.
        \step Die Streifen scharf in Öl anbraten und auf den Fladen verteilen.
        \step Die restlichen Beilagen auf die Fladen legen und zusammenrollen oder alles auf einem Teller anrichten.
    }
\end{recipe}
