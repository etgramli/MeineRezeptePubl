\begin{recipe}[
    preparationtime = {\unit[30]{Min}},
    bakingtime = {\unit[150]{Min}},
    portion = {\portion{4}},
    source = Etienne
]{Bœuf Bourgignon}\index{Boeuf Bourgignon@Bœuf Bourgignon}
    \graph{
        big=BoeufBourgignon_0
    }
    \ingredients{
        {\unit[1]{kg}}                   & Rindergulasch\\
        {\unit[2]{EL}}                   & Butter\\
        4                                & Zwiebeln\\
        {\unit[2]{EL}}                   & Tomatenmark\\
        {\unit[$\nicefrac{1}{2}$]{l}}    & Burgunder\\
        1                                & Lorbeerblatt\\
        {\unit[1]{Zweig}}                & Rosmarin\\
        4                                & Karotten\\
        {\unit[200]{g}}                  & Champignons\\
                                         & Salz\\
                                         & Pfeffer
    }
    \preparation{
        \step Das Fleisch scharf in der Butter anbraten. Zwiebeln schälen, achteln und andünsten.
        \step Das Tomatenmark dazu geben, kurz anbraten und mit dem Rotwein ablöschen.
        \step Lorbeerblatt und Rosmarin dazu geben und eine~Stunde köcheln lassen.
        \step Champignons putzen und vierteln, Karotten schälen und in Stücke schneiden und beides dazu geben.
        \step Karotten und Champignons dazu geben und 90~Minuten köcheln lassen.
        \step Abschmecken und servieren.
    }
\end{recipe}
