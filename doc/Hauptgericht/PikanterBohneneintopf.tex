    \ingredients{
        1                   & Knoblauchzehe\\
        {\unit[1]{EL}}      & Öl\\
        {\unit[1]{EL}}      & Mehl\\
        {\unit[1]{Dose}}    & Bohnen\\
        etwas               & Wasser\\
        {\unit[1]{TL}}      & Paprikapulver, süß\\
                            & Salz\\
                            & Pfeffer\\
        opt.                & Pfefferbeißer / Bierknacker
    }
    \preparation{
        \step Die Knoblauchzehe durch eine Presse drücken und mit dem Mehl in Öl andünsten.

    }
