\begin{recipe}[
    preparationtime = {\unit[20]{Min}},
    portion = {\portion{2}},
    source = Etienne
]{Spaghetti mit Fleischbällchen}\index{Spaghetti mit Fleischbaellchen@Spaghetti mit Fleischbällchen}
    \graph{
        big=SpaghettiMitFleischbaellchen_0
    }
    \ingredients{
        {\unit[200]{g}}    & Spaghetti\\
        {\unit[300]{g}}    & Hackfleisch, gemischt\\
        1                  & Zwiebel, klein\\
        {\unit[1]{TL}}     & Thymian\\
        {\unit[1]{TL}}     & Oregano, gerebelt\\
        {\unit[1]{TL}}     & Paprikapulver, mittelscharf\\
        {\unit[400]{g}}    & Tomaten, gehackt\\
                           & Salz\\
                           & Pfeffer
    }
    \preparation{
        \step Wasser zum Kochen bringen und die Spaghetti darin al dente zubereiten.
        \step Währenddessen die Zwiebel in feine Würfel schneiden und mit Hackfleisch und Gewürzen vermischen.
        \step Zu etwa 8 Bällchen formen und in einer Pfanne scharf von allen Seiten anbraten.
        \step Mit den Tomaten ablöschen und kurz einkochen lassen.
        \step Mit Salz und Pfeffer abschmecken und auf den Spaghetti servieren.
    }
\end{recipe}
