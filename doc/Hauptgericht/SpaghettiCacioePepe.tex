\begin{recipe}[
    preparationtime = {\unit[12]{Min}},
    portion = {\portion{2}},
    source = Etienne
]{Spaghetti Cacio e Pepe}\index{Spaghetti Cacio e Pepe}
    \graph{
        big=SpaghettiCacioePepe_0
    }
    \ingredients{
        {\unit[250]{g}}    & Spaghetti\\
        {\unit[75]{g}}     & Pecorino Romano\\
        {\unit[2]{EL}}     & Olivenöl\\
                           & Pfeffer, schwarz
    }
    \preparation{
        \step Spaghetti kochen und währenddessen den Pecorino hobeln. Etwas Kochwasser dazu geben und verrühren.
        \step Spaghetti abgießen und mit dem Pecorino vermischen, mit Pfeffer abschmecken und Olivenöl dazu geben.
    }
    \hint{%
        Kann man mit Scheiben von etwa 6 Oliven verfeinern.\newline
        Grüner Salat passt sehr gut dazu.
    }
\end{recipe}
