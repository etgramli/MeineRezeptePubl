\begin{recipe}[
    preparationtime = {\unit[30]{Min}},
    source = Etienne
]{Steak Diane mit Rosmarinkartoffeln}\index{Steak Diane mit Rosmarinkartoffeln}
    \graph{
        small=SteakDianeMitRosmarinkartoffeln_1,
        big=SteakDianeMitRosmarinkartoffeln_0
    }
    \ingredients{
        4                    & Roastbeef\\
        2                    & Schalotten\\
        {\unit[200]{g}}      & Champignons\\
        {\unit[100]{ml}}     & Rotwein\\
        {\unit[1]{EL}}       & Petersilie, gehackt\\
        {\unit[200]{ml}}     & Sahne\\
        6                    & Kartoffeln, mittel\\
        {\unit[1]{Zweig}}    & Rosmarin\\
                             & Öl\\
                             & Salz\\
                             & Pfeffer
    }
    \preparation{
        \step Kartoffeln in der Schale kochen, schälen und zur Seite stellen.
        \step Steaks mit einem Nudelholz etwas plätten, salzen und pfeffern. Scharf anbraten und beiseite stellen.
        \step Gewürfelte Zwiebel in der Pfanne glasig dünsten, Champignonscheiben dazu geben und kurz anbraten.
        \step Mit dem Rotwein ablöschen und Petersilie dazu geben und vermischen.
        \step Sahne dazu geben, mit Salz und Pfeffer abschmecken und die Steaks in die Sauce legen und ziehen lassen.
        \step Die Kartoffeln halbieren, mit der Schnittfläche auf eine Pfanne legen, salzen und Rosmarin darauf legen.
        \step Kartoffeln anbraten bis sie bräunen und mit den Steaks servieren.
    }
\end{recipe}
