\begin{recipe}[
    preparationtime = {\unit[20]{Min}},
    portion = {\portion{4}},
    source = Lili
]{Kartoffelbrei}\index{Kartoffelbrei}
    \graph{
        big=Kartoffelbrei_0
    }
    \ingredients{
        8                     & Kartoffeln, mittel\\
                              & Wasser\\
                              & Salz\\
        {\unit[1]{Schuss}}    & Sahne, süß\\
                              & Salz\\
                              & Muskat
    }
    \preparation{
        \step Kartoffeln schälen, in grobe Wüfel schneiden und in nicht zu viel Salzwasser gar kochen.
        \step Anschließend im Kochwasser zerstampfen und mit der süßen Sahne vermengen.
        \step Mit Salz und Muskat abschmecken.
    }
    \hint{%
        Kartoffelbrei für Neurodermitiker mit Milcheiweißallergie und zur Trennkost geeignet!
    }
\end{recipe}
