\begin{recipe}[
    preparationtime = {\unit[10]{Min}},
    portion = {\portion[Glas]{1}},
    source = Etienne
]{Eingelegte Chilis}\index{Eingelegte Chilis}
    \graph{
        big=EingelegteChili_0
    }
    \ingredients{
        7                   & Chilischoten\\
        {\unit[250]{ml}}    & Wasser\\
        {\unit[75]{ml}}     & Essig\\
        {\unit[25]{g}}      & Salz\\
                            & Olivenöl
    }
    \preparation{
        \step Wasser, Essig und Salz in einem Topf zum Kochen bringen.
        \step Die Chilischoten putzen, halbieren, die Kerne entfernen und klein schneiden.
        \step Die Schoten im Essigwasser {\unit[5]{Minuten}} abkochen, abtropfen und in ein abgekochtes Glas geben.
        \step Das Glas langsam mit Olivenöl füllen, sodass keine Luftbläschen übrig bleiben; kühl lagern.
    }
    \hint{%
        Lüftung einschalten! - Das Essig-Wasser reizt die Schleimhäute.
    }
\end{recipe}
