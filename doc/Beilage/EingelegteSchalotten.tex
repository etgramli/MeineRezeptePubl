\begin{recipe}[
    preparationtime = {\unit[15]{Min}},
    bakingtime = {\unit[40]{Min}},
    portion = {\portion[Glas]{1}},
    source = Etienne
]{Eingelegte Schalotten}\index{Eingelegte Schalotten}
    \graph{
        big=EingelegteSchalotten_0
    }
    \ingredients{
        {\unit[200]{g}}      & Schalotten\\
        {\unit[250]{ml}}     & Sherry, fino\\
        {\unit[50]{ml}}      & Whisky, sherryfassgereift\\
        {\unit[50]{ml}}      & Balsamico-Essig\\
        {\unit[1]{EL}}       & Oregano\\
        1                    & Lorbeerblatt\\
        {\unit[$\nicefrac{1}{2}$]{TL}}    & Zimt\\
        {\unit[1]{TL}}       & Chili-Salz\\
        {\unit[1]{Prise}}    & Pfeffer
    }
    \preparation{
        \step Den Schalotten bei geringer Hitze in Öl anbraten, bis sie von zwei Seiten braun sind.
        \step Mit Sherry, Whisky und Balsamico-Essig ablöschen, Gewürze dazu geben und 40 Minuten köcheln.
        \step Abkühlen lassen und in ein Glas abfüllen und kühl stellen.
    }
    \hint{%
        Kann püriert gut als Grundlage für Saucen verwendet werden.
    }
\end{recipe}
